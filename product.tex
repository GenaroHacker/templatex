\documentclass[12pt]{article}
\usepackage[a4paper, total={6in, 8in}]{geometry}
\usepackage{forloop}

\newcounter{ct}

% Define a command for drawing spaced lines
\newcommand{\lines}[2]{% #1: number of lines, #2: spacing between lines
  \forloop{ct}{1}{\value{ct} < #1}{\noindent\rule{\linewidth}{0.4pt}\\[#2]}
}

\begin{document}

\section*{Computer Text Section 1}
This section contains the first part of regular computer text.

\section*{Handwriting Section 1}
% Adjust line spacing for handwriting
{\openup 0.8cm
\lines{15}{8mm} % 15 lines with 8mm spacing
}

\section*{Computer Text Section 2}
This section contains the second part of regular computer text.

\section*{Handwriting Section 2}
% Continue with adjusted line spacing
{\openup 0.8cm
\lines{20}{8mm} % 20 lines with 8mm spacing
}

\end{document}
